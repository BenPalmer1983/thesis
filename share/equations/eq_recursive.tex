\newcommand{\eqFactorial}{
%% 
\begin{equation}
\begin{split}
f(n) = \left\{
\begin{array}{ll}
      1 & n = 1 \\
      n \times f(n-1) & n > 1 \\
\end{array} 
\right.
\end{split}
\label{eq:eqFactorial}
\end{equation}
}




\newcommand{\eqFibonacci}{
%% 
\begin{equation}
\begin{split}
f(n) = \left\{
\begin{array}{ll}
      1 & n \leq 2 \\
      f(n-1) + f(n-2) & n > 2 \\
\end{array} 
\right.
\end{split}
\label{eq:eqFibonacci}
\end{equation}
}
