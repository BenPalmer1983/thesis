    \begin{split}
      i \hbar \frac{\partial}{\partial t} \Psi \left( \vec{r},t\right) = \hat{H} \Psi \left( \vec{r},t\right)
    \end{split}
    \label{eq:eqSchrodingerTimeDependent}
  \end{equation}
