\newcommand{\eqGeneralArithmeticSeries}{
	%% Mathematical Methods for Physics and Engineering, Third Edition, Riley, Hobson, Bence
	%% Page	117
	\begin{equation}
	S_{N} = a + \left( a+d \right) + \left( a+2d \right) + \ldots + \left[a+\left(N-1\right)d\right] = \sum \limits_{0}^{N-1} \left( a + n d \right)
	\label{eq:eqGeneralArithmeticSeries}
	\end{equation}
}

\newcommand{\eqGeneralGeometicSeries}{
	%% Mathematical Methods for Physics and Engineering, Third Edition, Riley, Hobson, Bence
	%% Page	117
	\begin{equation}
	S_{N} = a + a r + a r^2 + \ldots + a r^{N-1} = \sum \limits_{0}^{N-1} \left( a r^n \right)
	\label{eq:eqGeneralGeometicSeries}
	\end{equation}
}

\newcommand{\eqComplexFourierSeries}{
	\begin{equation}
	f(t) = \sum \limits_{r=-\infty}^{\infty} c_r e^{i w_{r} t} \textnormal{   where   } w_{r} = \frac{2 \pi r}{T} 
	\label{eq:eqComplexFourierSeries}
	\end{equation}
}
