\newcommand{\eqEulersFormula}{
  \begin{equation}
    \begin{split}
      e^{i \theta} = cos \theta + i sin \theta
    \end{split}
    \label{eq:eqEulersFormula}
  \end{equation}
}



\newcommand{\eqStraightLine}{
  \begin{equation}
    \begin{split}
      f(x) = m x + c
    \end{split}
    \label{eq:eqStraightLine}
  \end{equation}
}


\newcommand{\eqTangentLine}{
  \begin{equation}
    \begin{split}
      f(x) = f'(x_n) (x - x_{n}) + f(x_{n})
    \end{split}
    \label{eq:eqTangentLine}
  \end{equation}
}
