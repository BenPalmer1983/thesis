\newcommand{\eqSimpsonsRule}{
  \begin{equation}
    \begin{split}
      \int _a ^b f(x) dx \approx \frac{b-a}{6} \left(f(a) + 4 f \left( \frac{a + b}{2} \right) + f(b) \right)
    \end{split}
    \label{eq:eqEulersFormula}
  \end{equation}
}
