\newcommand{\eqHybrid}{
%% Sample hybrid function
\begin{equation}
\begin{split}
f(x) = \left\{
\begin{array}{ll}
      0 & x\leq a \\
      \frac{x-a}{b-a} & a\leq x\leq b \\
      \frac{c-x}{c-b} & b\leq x\leq c \\
      1 & c\leq x \\
\end{array} 
\right.
\end{split}
\label{eq:eqHybrid}
\end{equation}
}
