\newcommand{\eqLaplaceTransform}{
  \begin{equation}
    F(s) = \int \limits_{0}^{\infty} f(t) \exp(-st) dt
    \label{eq:eqLaplaceTransform}
  \end{equation}
}

\newcommand{\eqFourierTransform}{
%% Mathematical Methods for Physics and Engineering, Third Edition, Riley, Hobson, Bence
%% Page	435
	\begin{equation}
	\tilde{f} \left( \omega \right) = \frac{1}{\sqrt{2 \pi}} \int_{-\infty}^{\infty} f \left( t \right) e^{-i \omega t} dt
	\label{eq:eqFourierTransform}
	\end{equation}
}

\newcommand{\eqInverseFourierTransform}{
%% Mathematical Methods for Physics and Engineering, Third Edition, Riley, Hobson, Bence
%% Page	435
	\begin{equation}
	f \left( t \right) = \frac{1}{\sqrt{2 \pi}} \int_{-\infty}^{\infty} \tilde{f} \left( \omega \right) e^{i \omega t} d \omega 
	\label{eq:eqInverseFourierTransform}
	\end{equation}
}
