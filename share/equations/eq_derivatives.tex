\newcommand{\eqNumericDerivative}{
  \begin{equation}
    \begin{split}
      f^{'}\left( x \right) \lim_{h \to 0} = \frac{f(x + h) - f(x-h)}{2h} 
    \end{split}
    \label{eq:eqNumericDerivative}
  \end{equation}
}

\newcommand{\eqNumericSDerivative}{
  \begin{equation}
    \begin{split}
      f^{''}\left( x \right) \lim_{h \to 0} = \frac{f(x + h) - 2f(x) + f(x - h) }{h^2}
    \end{split}
    \label{eq:eqNumericSDerivative}
  \end{equation}
}

\newcommand{\eqNumericTDerivative}{
  \begin{equation}
    \begin{split}
      f^{''}\left( x \right) \lim_{h \to 0} = \frac{f(x + 2h) - 2f(x + h) + 2f(x - h) - f(x - 2h) }{2 h^3}
    \end{split}
    \label{eq:eqNumericTDerivative}
  \end{equation}
}

\newcommand{\eqNumericTDerivativeB}{
  \begin{equation}
    \begin{split}
      f^{''}\left( x \right) \lim_{h \to 0} = \frac{f(x + 2h) - 3f(x + h) + 3f(x) - f(x - h) }{h^3}
    \end{split}
    \label{eq:eqNumericTDerivative}
  \end{equation}
}
